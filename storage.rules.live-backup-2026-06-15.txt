// BACKUP of the LIVE Storage rules as of 2026-06-15, BEFORE the recording-lockdown change.
// To roll back: copy everything below into Firebase Console → Storage → Rules → Publish.

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    match /voice_takes/{songId}/{filename} {
      allow read, write: if request.auth != null;
    }
    match /loop_tracks/{songId}/{filename} {
      allow read, write: if request.auth != null;
    }
    match /chunks/{recordingId}/{filename} {
      allow read, write: if request.auth != null;
    }
    match /shares/{songId}/{filename} {
      allow read;
      allow write: if request.auth != null;
    }
    match /{allPaths=**} {
      allow read, write: if false;
    }
  }
}
